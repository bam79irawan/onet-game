<script>
let abilities = {
  accuracy: 1,
  time: 60,
  skill: ['+', '-']
};

function generatePuzzle() {
  const container = document.getElementById('puzzle-container');
  container.innerHTML = '';
  const size = 4;
  for (let i = 0; i < size * size; i++) {
    const input = document.createElement('input');
    input.className = 'form-control puzzle-cell';
    input.setAttribute('type', 'number');
    input.setAttribute('placeholder', '?');
    container.appendChild(input);
  }
  alert('Isi kotak dengan angka yang hasil operasinya benar secara baris dan kolom!');
}

function upgrade(type) {
  if (type === 'accuracy') {
    abilities.accuracy += 1;
    alert('Akurasi meningkat ke level ' + abilities.accuracy);
  } else if (type === 'time') {
    abilities.time += 10;
    alert('Waktu tambahan: ' + abilities.time + ' detik');
  } else if (type === 'skill') {
    if (!abilities.skill.includes('*')) {
      abilities.skill.push('*');
    } else if (!abilities.skill.includes('/')) {
      abilities.skill.push('/');
    } else {
      alert('Semua operasi sudah dibuka!');
      return;
    }
    alert('Operasi sekarang: ' + abilities.skill.join(', '));
  }
}
</script>
